                    from turtle import Turtle

                    def curvedCircles(turtle, angle, length, radius):
                        turtle.color("orange")
                        turtle.penup()
                        turtle.right(angle)
                        turtle.forward(length)
                        turtle.pendown()
                        turtle.circle(radius)
                        length = length - 5
                        angle = angle + 5
                        radius = radius - 5
                        if (radius > 5):
                            curvedCircles(turtle, angle, length, radius)

                    def circleSun(turn):
                        animationSpeed = 0
                        tim = Turtle()
                        tim.speed(animationSpeed)
                        for i in range(20):
                            tim.penup()
                            tim.setx(0)
                            tim.sety(0)
                            tim.right(turn)
                            tim.pendown()
                            curvedCircles(tim, 0, 50, 50)

                    circleSun(10)
